Templates



Quando você cria uma função normal em C++, ela é específica para um tipo de dado. Por exemplo, uma função que soma dois números inteiros não funcionará diretamente para números decimais. Aqui é onde entram os templates de funções.

Com templates de funções, você pode criar uma função que funciona para diferentes tipos de dados. Por exemplo: 


                        
                        #include <iostream>
                        using namespace std;
                                
                        template<typename   NewType>
                        void  imprimi( NewType numero1,  NewType numero2,  NewType numero3,  NewType numero4);
                                
                                
                        template<typename   NewType>
                        void  imprimi( NewType numero1,  NewType numero2,  NewType numero3);
                        
                                
                        template<typename   NewType> 
                        void  imprimi( NewType numero1,  NewType numero2);
                                
                                
                        template<typename   NewType>
                        void  imprimi( NewType numero1);
                                
                        int main()
                        {
                            cout  <<  "Hello REAA!"  <<  endl;
                             imprimi(99, 44, 3, 7);
                             imprimi(1, 8, 37);
                             imprimi(17, 66);
                             imprimi(1025);
                                
                                
                                
                                
                            return 0;
                        }
                                
                                
                                
                        template<typename   NewType>
                        void  imprimi( NewType numero1,  NewType numero2,  NewType numero3,  NewType numero4)
                        {
                            cout << "imprime: " << numero1 << '-' << numero2 << '-' << numero3 << '-' << numero4 << endl;
                        }
                                
                                
                        template<typename   NewType>
                        void  imprimi( NewType numero1,  NewType numero2,  NewType numero3)
                        {
                            cout << "imprime: " << numero1 << '-' << numero2 << '-' << numero3 << endl;
                        }
                                
                                
                        template<typename   NewType>  
                        void  imprimi( NewType numero1,  NewType numero2)
                        {
                            cout << "imprime: " << numero1 << '-' << numero2 << endl;
                        }
                                
                                
                        template<typename   NewType> 
                        void  imprimi( NewType numero1)
                        {
                            cout << "imprime: " << numero1 << endl;
                        }
